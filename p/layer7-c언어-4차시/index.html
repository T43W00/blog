#include <stdio.h>

void hanoi(int n, int from, int via, int to) {
    
    if (n == 1) {
        printf("%d %d\n", from, to);
        return;
    }
    
    hanoi(n - 1, from, to, via);
    printf("%d %d\n", from, to);
    hanoi(n - 1, via, from, to);
}

        sum *= 2;
    }

    printf("%d\n", sum - 1);

    hanoi(n, 1, 2, 3);

    return 0;
}

#include <stdio.h>

int parent[100001];

int find(int x) {
    if (parent[] == x) return x;
    return parent[x] = find(parent[x]);
}

void Union(int a, int b) {
    a = find(a);
    b = find(b);
    parent[a] = b;
}

int main() {
    int G, P, gi;
    scanf("%d %d", &G, &P);

    for (int i = 0; i <= G; i++) {
        parent[i] = i;
    }

    int cnt = 0;
    for (int i = 0; i < P; i++) {
        scanf("%d", &gi);

        int D = find(gi);
        if (D == 0) break;

        Union(D, D - 1);
        cnt++;
    }

    printf("%d\n", cnt);

    return 0;
}
