        call read
        call str_to_int
        call main_
        mov eax, 0
        ret

        mov eax, 3
        mov ebx, 0
        mov ecx, num
        mov edx, 4

        mov esi, num

        ret

        movzx edx, byte [esi]

        add ecx, edx
        inc esi

        jne .loop
        ret

        xor eax, eax
        call .loop
        ret

        push ecx
        inc eax
        mov ecx, eax
        push eax
        call .loop_
        call print_newline
        pop eax
        pop ecx
        loop .loop
        ret

.loop_:
        push ecx
        call print_star
        pop ecx
        loop .loop_
        ret

        mov eax, 4
        mov ebx, 1
        mov ecx, star
        mov edx, 1
        int 0x80
        ret

        mov eax, 4
